题目

给定一张带权有向无环图，像神经网络一样前向传播计算，输出最终结果。计算的方式为
Ci=∑(j,i)∈EWjiCj−UiC_i= \sum_{(j,i)\in E}W_{ji}C_j-U_iCi​=(j,i)∈E∑​Wji​Cj​−Ui​
除输入层外

来源：P1038 [NOIP2003 提高组] 神经网络 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)

解答

计算的逻辑并不难实现，但计算要满足一定顺序，对于上图中，1，2，3为输入层，6为输出层，应要先从1，2，3更新4，5的值，再根据4，5的值计算6。即对应DAG的拓扑序。

拓扑排序通过查找入度为0并更新来完成，剩下的就是模拟了。

#include <bits/stdc++.h>
using namespace std;

const int N = 5e5;

struct DiGraph {
    struct edge {
        int next;
        double w;
        edge(int next, double w) : next(next), w(w) {}
    };
    vector<vector<edge>> vex;
    vector<int> in;
    DiGraph(int N) : vex(N), in(N, 0) {}
    void insert(int pre, int next, double weight = 0) {
        vex[pre].emplace_back(next, weight);
        in[next]++;
    }
    vector<int> toposort(int n) {
        auto t = in;
        vector<int> arr;
        while (arr.size() < n) {
            for (int i = 1; i <= n; i++) {
                if (t[i] == 0) {
                    arr.push_back(i);
                    for (auto &p : vex[i]) t[p.next]--;
                    t[i]--;
                }
            }
        }
        return arr;
    }
} dig(N);

int n, p;
double arr[N][2];

int main() {
    cin >> n >> p;
    for (int i = 1; i <= n; i++) {
        cin >> arr[i][0];
        cin >> arr[i][1];
        if (arr[i][0]) arr[i][1] = 0;
    }
    int a, b;
    double c;
    for (int i = 0; i < p; i++) {
        cin >> a >> b >> c;
        dig.insert(a, b, c);
    }

    auto m = dig.toposort(n);
    for (auto &p : m) {
        if (arr[p][0] > arr[p][1])
            for (auto &t : dig.vex[p]) arr[t.next][0] += (arr[p][0] - arr[p][1]) * t.w;
    }

    int f = 1;
    for (int i = 1; i <= n; i++) {

    }

    if (f) cout << "NULL";

    return 0;
}

来源：P1106 删数问题 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)

解答

经典的贪心问题。

只考虑一次操作，从左往右找，若相邻两数右边的比左边的小，则应删去左边的数。这使得新数在此处的数值是最小的。

如此反复kkk次删除，实际上使得该数的数位从左往右保持单调增。

若并没有删除kkk次，按照单调增，删去最右边的数使得新数最小。

#include <bits/stdc++.h>
using namespace std;
string a;
int n;

int main() {
    cin >> a >> n;
    int f;
    while (n) {
        while (a[0] == '0') a = a.substr(1);
        f = 1;
        for (int i = 0; i < a.size(); i++) {
            if (a[i - 1] > a[i]) {
                a = a.substr(0, i - 1) + a.substr(i);
                n--;
                f = 0;
                break;
            }
        }
        if(f) break;
    }
    cout << a.substr(0, a.size() - n);
    return 0;
}

LG_P1434 滑雪

文件夹：solve/LG_P1434

题目

给定一个二维数组，记一条路径为正交方向上移动若干格的轨迹(不重叠)，求最长的单调下降的路径的长度

来源：P1434 [SHOI2002]滑雪 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)

解答

路径可以搜索，以一点为起点，dfs搜索直到没有可下降的方向，得出以该点为起点的最长路径。每个节点都要搜索一遍，时间复杂度很大。

题目要求路径上的值是单调下降的，当搜索到一点时，之前的路径不会和从这一点开始搜索的路径重叠，这样就可以把从该点出发得到的路径最大值记录下来，搜索到这一点直接计算即可，也就是记忆化搜索。

记忆化搜索和动态规划是等价的，所以本题也可以用动态规划的写法。

#include <bits/stdc++.h>
using namespace std;
int n, m;
int a[101][101][2]{};
int ans = 0;
int dir[4][2] = {-1, 0, 1, 0, 0, -1, 0, 1};

void fun(int i, int j) {
    if (a[i][j][1] != 0) return;
    // cout<<i<<" "<<j<<endl;
    int x, y, f = 0;
    for (int k = 0; k < 4; k++) {
        x = i + dir[k][0], y = j + dir[k][1];
        if (x >= 0 && y >= 0 && x < n && y < m && a[x][y][0] < a[i][j][0]) {
            f++;
            if (a[x][y][1] == 0) fun(x, y);
            a[i][j][1] = max(a[x][y][1] + 1, a[i][j][1]);
        }
        ans = max(a[i][j][1], ans);
    }
    if (f == 0) {
        a[i][j][1] = 1;
        ans = max(a[i][j][1], ans);
    }
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; ++j) cin >> a[i][j][0];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; ++j) fun(i, j);
    cout << ans;
    return 0;
}

  1. iii的子节点只能由iii向其传递消息；

我想到的节省时间的方法是，对于初始知道消息的节点(记为根节点)，则将其一个子节点提升为根节点，则只要再重新计算原根节点和提升的子节点的值即可，也是可以递归求解的。

#include <bits/stdc++.h>

    };

    void insert(int i, int j, int v) { vex[i].emplace_back(j, v), vex[j].emplace_back(i, v); }

    void dfs(int start) {
        vector<int> t(N, 0);
        arr[start].idx = start;
        arr[start].val = dfs(start, t);
    }

    int dfs(int s, auto &f) {
        f[s] = 1;
        vector<int> t;

            if (!f[e.next]) {
                t.push_back(dfs(e.next, f));
            }
        }
        sort(t.begin(), t.end());
        int g = 0;
        for (int i = 1; i <= t.size(); i++) {
            g = max(t[t.size() - i] + i, g);
        }
        return g;
    }
} tree;

排列的逆序数定义的计算方法是：

记 p(i)p(i)p(i) 为排列中第 iii 个数和其前面 i−1i-1i−1 个数不满足某一偏序关系的个数，该排列的逆序数为 p(i)p(i)p(i) 求和，时间复杂度为 O(n2)O(n^2)O(n2)；

O(nlogn)O(nlogn)O(nlogn) 的方法有几种，这里利用树状数组快速计算不断更新的前缀和。

将排列从大到小依次放入对应位置，其位置前面已放置的数大于它本身。若用111表示已放置，000表示未放置，则放置数前一个位置的部分和即为p(i)p(i)p(i)，数组的前缀和用树状数组维护，更新和计算的复杂度均为O(logn)O(logn)O(logn)，放置n次，复杂度为O(nlogn)O(nlogn)O(nlogn)
